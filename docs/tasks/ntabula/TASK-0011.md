```swift
import XCTest
@testable import nTabula

final class NotionServiceTests: XCTestCase {
    var sut: NotionService!
    var mockSession: URLSession!

    override func setUp() {
        super.setUp()
        mockSession = URLSession.makeMockSession()
        sut = NotionService(token: "test-token", session: mockSession)
    }

    override func tearDown() {
        MockURLProtocol.requestHandler = nil
        sut = nil
        mockSession = nil
        super.tearDown()
    }
}
```

### 2. fetchDatabases 正常系テスト 🔵

**信頼性**: 🔵 *REQ-404・NotionModels.swift NotionDatabaseより*

```swift
func testFetchDatabases_success() async throws {
    let mockJSON = """
    {
        "results": [
            {
                "id": "db-id-1",
                "title": [{"type": "text", "plain_text": "My DB"}],
                "properties": {
                    "Name": {"type": "title"}
                }
            }
        ],
        "has_more": false,
        "next_cursor": null
    }
    """

    MockURLProtocol.requestHandler = { request in
        XCTAssertTrue(request.url?.absoluteString.contains("search") ?? false)
        XCTAssertEqual(request.value(forHTTPHeaderField: "Notion-Version"), "2022-06-28")

        let response = HTTPURLResponse(
            url: request.url!,
            statusCode: 200,
            httpVersion: nil,
            headerFields: nil
        )!
        return (response, mockJSON.data(using: .utf8)!)
    }

    let databases = try await sut.fetchDatabases()

    XCTAssertEqual(databases.count, 1)
    XCTAssertEqual(databases.first?.id, "db-id-1")
    XCTAssertEqual(databases.first?.titlePropertyName, "Name")
}
```

### 3. NotionAPIError デコードテスト 🔵

**信頼性**: 🔵 *EDGE-001 NotionAPIError処理より*

```swift
func testFetchDatabases_notionAPIError() async throws {
    let errorJSON = """
    {
        "object": "error",
        "status": 401,
        "code": "unauthorized",
        "message": "API token is invalid."
    }
    """

    MockURLProtocol.requestHandler = { request in
        let response = HTTPURLResponse(
            url: request.url!,
            statusCode: 401,
            httpVersion: nil,
            headerFields: nil
        )!
        return (response, errorJSON.data(using: .utf8)!)
    }

    do {
        _ = try await sut.fetchDatabases()
        XCTFail("エラーがスローされるべき")
    } catch let error as NotionAPIError {
        XCTAssertEqual(error.code, "unauthorized")
        XCTAssertEqual(error.status, 401)
        XCTAssertTrue(error.errorDescription?.contains("unauthorized") ?? false)
    }
}
```
