### 1. テストセットアップ 🔵

**信頼性**: 🔵 *NFR-302 UserDefaults注入・dataflow.md フロー11より*

**実装ファイル**: `nTabulaTests/PersistenceManagerTests.swift`

```swift
import XCTest
@testable import nTabula

final class PersistenceManagerTests: XCTestCase {
    var sut: PersistenceManager!
    var testDefaults: UserDefaults!

    override func setUp() {
        super.setUp()
        // テストごとに独立した UserDefaults を使用
        testDefaults = UserDefaults(suiteName: "test-\(UUID())")
        sut = PersistenceManager(defaults: testDefaults)
    }

    override func tearDown() {
        sut = nil
        testDefaults.removePersistentDomain(forName: testDefaults.description)
        testDefaults = nil
        super.tearDown()
    }
}
```

### 2. タブ保存・読み込みテスト 🔵

**信頼性**: 🔵 *persistence-schema.md TabItem JSONスキーマより*

```swift
func testSaveAndLoadTabs() throws {
    let tab = TabItem(
        id: UUID(),
        title: "テスト",
        content: "# Hello",
        notionPageID: nil,
        databaseID: nil,
        titlePropertyName: "Name",
        isPinned: false,
        isDirty: false,
        createdAt: Date(),
        updatedAt: Date()
    )

    sut.saveTabs([tab])
    let loaded = sut.loadTabs()

    XCTAssertEqual(loaded.count, 1)
    XCTAssertEqual(loaded.first?.id, tab.id)
    XCTAssertEqual(loaded.first?.title, "テスト")
    XCTAssertEqual(loaded.first?.content, "# Hello")
    XCTAssertFalse(loaded.first?.isDirty ?? true, "isDirty は永続化しない（起動時常にfalse）")
}
```

### 3. フォントサイズ EDGE-104 テスト 🔵

**信頼性**: 🔵 *EDGE-104 フォントサイズ0以下のfallbackより*

```swift
func testLoadFontSizeDefaultsTo14WhenZeroOrNegative() {
    // 0 を保存
    sut.saveFontSize(0)
    XCTAssertEqual(sut.loadFontSize(), 14, "0以下は14ptにフォールバック")

    // 負値を保存
    sut.saveFontSize(-1)
    XCTAssertEqual(sut.loadFontSize(), 14, "負値は14ptにフォールバック")

    // 正常値
    sut.saveFontSize(18)
    XCTAssertEqual(sut.loadFontSize(), 18)
}
```
